<script>

const ctx = new (window.AudioContext || window.webkitAudioContext)();


let osc, gain, filter;

let isOn = false;

let mouthOpen = false;

let currentFreq = 440;


// scale

const scale = [

130.81,146.83,164.81,174.61,196.00,220.00,246.94,

261.63,293.66,329.63,349.23,392.00,440.00,493.88

];


// START

function startSound() {

  osc = ctx.createOscillator();

  gain = ctx.createGain();

  filter = ctx.createBiquadFilter();


  osc.type = "sawtooth";

  gain.gain.value = 0.15;

  filter.type = "lowpass";


  osc.connect(filter);

  filter.connect(gain);

  gain.connect(ctx.destination);


  osc.start();

  setFrequency(currentFreq);

}


// STOP

function stopSound() {

  if (osc) osc.stop();

}


// GLIDE

function setFrequency(freq) {

  currentFreq = freq;

  if (!osc) return;


  osc.frequency.cancelScheduledValues(ctx.currentTime);

  osc.frequency.setTargetAtTime(freq, ctx.currentTime, 0.08);

}


// BUTTON SET

function setFreq(freq) {

  setFrequency(freq);

}


// +/- adjust

function adjustFreq(amount) {

  setFrequency(currentFreq + amount);

}


// POWER BUTTON

document.getElementById("powerBtn").onclick = () => {

  if (!isOn) {

    startSound();

    isOn = true;

    powerBtn.classList.add("active");

  } else {

    stopSound();

    isOn = false;

    powerBtn.classList.remove("active");

  }

};


// 🎸 FRETBOARD

const fretCanvas = document.getElementById("fretboard");

const fctx = fretCanvas.getContext("2d");


function drawFretboard(active = -1) {

  fctx.clearRect(0,0,300,100);

  let w = 300 / scale.length;


  scale.forEach((note,i)=>{

    fctx.fillStyle = (i===active) ? "#22c55e" : "#334155";

    fctx.fillRect(i*w,0,w-2,100);

  });

}

drawFretboard();


// PLAY FROM MOUSE

function playFromX(x) {

  const i = Math.floor((x/window.innerWidth)*scale.length);

  const idx = Math.max(0, Math.min(scale.length-1, i));


  setFrequency(scale[idx]);

  drawFretboard(idx);

}


// MOUSE

window.addEventListener("mousemove", e=>{

  if (!isOn) return;

  playFromX(e.clientX);

});


// TOUCH

window.addEventListener("touchmove", e=>{

  if (!isOn) return;

  playFromX(e.touches[0].clientX);

});


// 😮 FILTER

function updateFilter() {

  if (!filter) return;

  if (mouthOpen) {

    filter.frequency.setTargetAtTime(2000, ctx.currentTime, 0.05);

  } else {

    filter.frequency.setTargetAtTime(400, ctx.currentTime, 0.05);

  }

}


// KEY

window.addEventListener("keydown", e=>{

  if (e.code==="Space") {

    mouthOpen = true;

    updateFilter();

  }

});

window.addEventListener("keyup", e=>{

  if (e.code==="Space") {

    mouthOpen = false;

    updateFilter();

  }

});
